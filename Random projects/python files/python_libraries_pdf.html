1. os - Operating System Interface

Provides functions for interacting with the operating system.

Basic Syntax

import os

# Directory operations
os.getcwd()                    # Get current working directory
os.chdir(path)                 # Change directory
os.listdir(path)               # List directory contents
os.mkdir(path)                 # Create directory
os.makedirs(path)              # Create nested directories
os.rmdir(path)                 # Remove directory

# File operations
os.remove(path)                # Delete file
os.rename(old, new)            # Rename file/directory

# Path operations
os.path.join(path1, path2)     # Join paths
os.path.exists(path)           # Check if path exists
os.path.isfile(path)           # Check if file
os.path.isdir(path)            # Check if directory
os.path.basename(path)         # Get filename
os.path.dirname(path)          # Get directory name
os.path.abspath(path)          # Get absolute path

Example Problem 1: List all .txt files

import os

# List all .txt files in current directory
txt_files = [f for f in os.listdir('.') if f.endswith('.txt')]
print(f"Found {len(txt_files)} text files: {txt_files}")

Example Problem 2: Create directory structure

import os

# Create nested directories
project_structure = ['data', 'src', 'tests', 'docs']
for folder in project_structure:
    if not os.path.exists(folder):
        os.makedirs(folder)
        print(f"Created: {folder}")

Example Problem 3: Find files by extension

import os

def find_files(directory, extension):
    found = []
    for root, dirs, files in os.walk(directory):
        for file in files:
            if file.endswith(extension):
                found.append(os.path.join(root, file))
    return found

python_files = find_files('.', '.py')
print(f"Python files: {len(python_files)}")

Example Problem 4: File size calculator

import os

def get_directory_size(path):
    total = 0
    for entry in os.scandir(path):
        if entry.is_file():
            total += entry.stat().st_size
        elif entry.is_dir():
            total += get_directory_size(entry.path)
    return total

size = get_directory_size('.')
print(f"Directory size: {size / 1024:.2f} KB")

3. datetime - Date and Time Handling

Classes for manipulating dates and times.

Basic Syntax

from datetime import datetime, date, time, timedelta

# Current date/time
datetime.now()                 # Current datetime
date.today()                   # Current date
time()                         # Time object

# Create specific datetime
datetime(2024, 1, 15, 14, 30)  # Year, month, day, hour, minute
date(2024, 1, 15)              # Year, month, day

# Time arithmetic
timedelta(days=7, hours=2)     # Time difference

# String conversion
datetime.strptime(string, format)   # String to datetime
datetime.strftime(format)           # Datetime to string

# Common format codes:
# %Y - Year (4 digits)    %m - Month (01-12)    %d - Day (01-31)
# %H - Hour (00-23)       %M - Minute (00-59)   %S - Second (00-59)

Example Problem 1: Days until birthday

from datetime import datetime, date

def days_until_birthday(birth_month, birth_day):
    today = date.today()
    next_birthday = date(today.year, birth_month, birth_day)
    
    if next_birthday < today:
        next_birthday = date(today.year + 1, birth_month, birth_day)
    
    return (next_birthday - today).days

days = days_until_birthday(12, 25)
print(f"Days until birthday: {days}")

Example Problem 2: Age calculator

from datetime import date

def calculate_age(birth_date):
    today = date.today()
    age = today.year - birth_date.year
    
    # Adjust if birthday hasn't occurred this year
    if (today.month, today.day) < (birth_date.month, birth_date.day):
        age -= 1
    
    return age

birthdate = date(1990, 5, 15)
print(f"Age: {calculate_age(birthdate)} years")

Example Problem 3: Date formatting

from datetime import datetime

now = datetime.now()

# Different formats
print(now.strftime("%Y-%m-%d"))           # 2026-01-24
print(now.strftime("%B %d, %Y"))          # January 24, 2026
print(now.strftime("%I:%M %p"))           # 02:30 PM
print(now.strftime("%A, %B %d, %Y"))      # Saturday, January 24, 2026

Example Problem 4: Business days calculator

from datetime import datetime, timedelta

def add_business_days(start_date, days):
    current = start_date
    while days > 0:
        current += timedelta(days=1)
        # Skip weekends (5=Saturday, 6=Sunday)
        if current.weekday() < 5:
            days -= 1
    return current

start = datetime(2026, 1, 24)
result = add_business_days(start, 10)
print(f"10 business days later: {result.strftime('%Y-%m-%d')}")

4. random - Random Number Generation

Generate pseudo-random numbers and make random selections.

Basic Syntax

import random

# Random numbers
random.random()                # Float [0.0, 1.0)
random.uniform(a, b)           # Float [a, b]

random.choices(population, k=n)  # n elements (with replacement)

# Shuffling

# Seeding
random.seed(n)                 # Set random seed for reproducibility

Example Problem 4: Random team selector

import random

def create_teams(players, team_size):
    shuffled = players.copy()
    random.shuffle(shuffled)
    
    teams = []
    for i in range(0, len(shuffled), team_size):
        teams.append(shuffled[i:i + team_size])
    return teams

players = ['Alice', 'Bob', 'Charlie', 'David', 'Eve', 'Frank']

for i, team in enumerate(teams, 1):
    print(f"Team {i}: {team}")

Example Problem 3: Compound interest

import math

def compound_interest(principal, rate, time, n=12):
    """
    principal: initial amount
    rate: annual interest rate (decimal)
    time: years
    n: compounds per year
    """
    amount = principal * math.pow(1 + rate/n, n*time)
    return round(amount, 2)

result = compound_interest(1000, 0.05, 10)
print(f"$1000 at 5% for 10 years: ${result}")

Example Problem 4: Triangle solver

import math

def solve_triangle(a, b, angle_c_degrees):
    """Calculate third side using law of cosines"""
    angle_c = math.radians(angle_c_degrees)
    c = math.sqrt(a**2 + b**2 - 2*a*b*math.cos(angle_c))
    
    # Calculate other angles using law of sines
    angle_a = math.asin(a * math.sin(angle_c) / c)
    angle_b = math.pi - angle_a - angle_c
    
    return {
        'side_c': round(c, 2),
        'angle_a': round(math.degrees(angle_a), 2),
        'angle_b': round(math.degrees(angle_b), 2)
    }

result = solve_triangle(5, 7, 60)
print(f"Triangle solution: {result}")

Example Problem 2: Configuration manager

import json

class ConfigManager:
    def __init__(self, filename='config.json'):
        self.filename = filename
        self.config = self.load()
    
    def load(self):
        try:
            with open(self.filename, 'r') as f:
                return json.load(f)
        except FileNotFoundError:
            return {}
    
    def save(self):
        with open(self.filename, 'w') as f:
            json.dump(self.config, f, indent=2)
    
    def get(self, key, default=None):
        return self.config.get(key, default)
    
    def set(self, key, value):
        self.config[key] = value
        self.save()

config = ConfigManager()
config.set('theme', 'dark')
print(f"Theme: {config.get('theme')}")

import json
import csv

def csv_to_json(csv_file, json_file):
    data = []
    with open(csv_file, 'r') as f:
        reader = csv.DictReader(f)
        for row in reader:
            data.append(row)
    
    with open(json_file, 'w') as f:
        json.dump(data, f, indent=2)
    
    return len(data)

7. re - Regular Expressions

Pattern matching with regular expressions.

Basic Syntax

import re

# Search functions
re.search(pattern, string)     # Search anywhere
re.match(pattern, string)      # Match at beginning
re.fullmatch(pattern, string)  # Match entire string
re.findall(pattern, string)    # Find all occurrences
re.finditer(pattern, string)   # Iterator of matches

# Substitution
re.sub(pattern, repl, string)  # Replace matches
re.subn(pattern, repl, string) # Replace and count

# Split
re.split(pattern, string)      # Split by pattern

# Common patterns:
# .  - Any character         \d - Digit         \w - Word character
# \s - Whitespace            ^  - Start         $  - End
# *  - 0 or more             +  - 1 or more     ? - 0 or 1
# [] - Character class       () - Group         | - OR

Example Problem 1: Email extractor

import re

def extract_emails(text):
    pattern = r'\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\.[A-Z|a-z]{2,}\b'
    return re.findall(pattern, text)

text = """
Contact us at info@example.com or support@test.org
Sales: sales@company.co.uk
"""

emails = extract_emails(text)
print(f"Found emails: {emails}")

Example Problem 2: Phone number validator

import re

def validate_phone(number):
    # Matches: (123) 456-7890, 123-456-7890, 1234567890
    patterns = [
        r'^\(\d{3}\)\s\d{3}-\d{4}$',
        r'^\d{3}-\d{3}-\d{4}$',
        r'^\d{10}$'
    ]
    
    return any(re.match(pattern, number) for pattern in patterns)

phones = ['(555) 123-4567', '555-123-4567', '5551234567', 'invalid']
for phone in phones:
    print(f"{phone}: {validate_phone(phone)}")

Example Problem 3: Password strength checker

import re

def check_password_strength(password):
    if len(password) < 8:
        return "Weak: Too short"
    
    checks = {
        'uppercase': r'[A-Z]',
        'lowercase': r'[a-z]',
        'digit': r'\d',
        'special': r'[!@#$%^&*(),.?":{}|<>]'
    }
    
    score = sum(bool(re.search(pattern, password)) 
                for pattern in checks.values())
    
    if score == 4:
        return "Strong"
    elif score >= 2:
        return "Medium"
    else:
        return "Weak"

passwords = ['pass', 'Password1', 'P@ssw0rd!']
for pwd in passwords:
    print(f"{pwd}: {check_password_strength(pwd)}")
